113 void WalletModel::pollBalanceChanged()
114 {
115  // Get required locks upfront. This avoids the GUI from getting stuck on
116  // periodical polls if the core is holding the locks for a longer time -
117  // for example, during a wallet rescan.
118  TRY_LOCK(cs_main, lockMain);
119  if(!lockMain)
120  return;
121  TRY_LOCK(wallet->cs_wallet, lockWallet);
122  if(!lockWallet)
123  return;
124 
125  if(fForceCheckBalanceChanged || chainActive.Height() != cachedNumBlocks)
126  {
127  fForceCheckBalanceChanged = false;
128 
129  // Balance and number of transactions might have changed
130  cachedNumBlocks = chainActive.Height();
131 
132  checkBalanceChanged();
133  if(transactionTableModel)
134  transactionTableModel->updateConfirmations();
135  }
136 }

192 WalletModel::SendCoinsReturn WalletModel::prepareTransaction(WalletModelTransaction &transaction, const CCoinControl *coinControl)
193 {
194  CAmount total = 0;
195  bool fSubtractFeeFromAmount = false;
196  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
197  std::vector<CRecipient> vecSend;
198 
199  if(recipients.empty())
200  {
201  return OK;
202  }
203 
204  QSet<QString> setAddress; // Used to detect duplicates
205  int nAddresses = 0;
206 
207  // Pre-check input data for validity
208  Q_FOREACH(const SendCoinsRecipient &rcp, recipients)
209  {
210  if (rcp.fSubtractFeeFromAmount)
211  fSubtractFeeFromAmount = true;
212 
213  if (rcp.paymentRequest.IsInitialized())
214  { // PaymentRequest...
215  CAmount subtotal = 0;
216  const payments::PaymentDetails& details = rcp.paymentRequest.getDetails();
217  for (int i = 0; i < details.outputs_size(); i++)
218  {
219  const payments::Output& out = details.outputs(i);
220  if (out.amount() <= 0) continue;
221  subtotal += out.amount();
222  const unsigned char* scriptStr = (const unsigned char*)out.script().data();
223  CScript scriptPubKey(scriptStr, scriptStr+out.script().size());
224  CAmount nAmount = out.amount();
225  CRecipient recipient = {scriptPubKey, nAmount, rcp.fSubtractFeeFromAmount};
226  vecSend.push_back(recipient);
227  }
228  if (subtotal <= 0)
229  {
230  return InvalidAmount;
231  }
232  total += subtotal;
233  }
234  else
235  { // User-entered bitcoin address / amount:
236  if(!validateAddress(rcp.address))
237  {
238  return InvalidAddress;
239  }
240  if(rcp.amount <= 0)
241  {
242  return InvalidAmount;
243  }
244  setAddress.insert(rcp.address);
245  ++nAddresses;
246 
247  CScript scriptPubKey = GetScriptForDestination(CBitcoinAddress(rcp.address.toStdString()).Get());
248  CRecipient recipient = {scriptPubKey, rcp.amount, rcp.fSubtractFeeFromAmount};
249  vecSend.push_back(recipient);
250 
251  total += rcp.amount;
252  }
253  }
254  if(setAddress.size() != nAddresses)
255  {
256  return DuplicateAddress;
257  }
258 
259  CAmount nBalance = getBalance(coinControl);
260 
261  if(total > nBalance)
262  {
263  return AmountExceedsBalance;
264  }
265 
266  {
267  LOCK2(cs_main, wallet->cs_wallet);
268 
269  transaction.newPossibleKeyChange(wallet);
270 
271  CAmount nFeeRequired = 0;
272  int nChangePosRet = -1;
273  std::string strFailReason;
274 
275  CWalletTx *newTx = transaction.getTransaction();
276  CReserveKey *keyChange = transaction.getPossibleKeyChange();
277  bool fCreated = wallet->CreateTransaction(vecSend, *newTx, *keyChange, nFeeRequired, nChangePosRet, strFailReason, coinControl);
278  transaction.setTransactionFee(nFeeRequired);
279  if (fSubtractFeeFromAmount && fCreated)
280  transaction.reassignAmounts(nChangePosRet);
281 
282  if(!fCreated)
283  {
284  if(!fSubtractFeeFromAmount && (total + nFeeRequired) > nBalance)
285  {
286  return SendCoinsReturn(AmountWithFeeExceedsBalance);
287  }
288  Q_EMIT message(tr("Send Coins"), QString::fromStdString(strFailReason),
289  CClientUIInterface::MSG_ERROR);
290  return TransactionCreationFailed;
291  }
292 
293  // reject absurdly high fee. (This can never happen because the
294  // wallet caps the fee at maxTxFee. This merely serves as a
295  // belt-and-suspenders check)
296  if (nFeeRequired > maxTxFee)
297  return AbsurdFee;
298  }
299 
300  return SendCoinsReturn(OK);
301 }

303 WalletModel::SendCoinsReturn WalletModel::sendCoins(WalletModelTransaction &transaction)
304 {
305  QByteArray transaction_array; /* store serialized transaction */
306 
307  {
308  LOCK2(cs_main, wallet->cs_wallet);
309  CWalletTx *newTx = transaction.getTransaction();
310 
311  Q_FOREACH(const SendCoinsRecipient &rcp, transaction.getRecipients())
312  {
313  if (rcp.paymentRequest.IsInitialized())
314  {
315  // Make sure any payment requests involved are still valid.
316  if (PaymentServer::verifyExpired(rcp.paymentRequest.getDetails())) {
317  return PaymentRequestExpired;
318  }
319 
320  // Store PaymentRequests in wtx.vOrderForm in wallet.
321  std::string key("PaymentRequest");
322  std::string value;
323  rcp.paymentRequest.SerializeToString(&value);
324  newTx->vOrderForm.push_back(make_pair(key, value));
325  }
326  else if (!rcp.message.isEmpty()) // Message from normal bitcoin:URI (bitcoin:123...?message=example)
327  newTx->vOrderForm.push_back(make_pair("Message", rcp.message.toStdString()));
328  }
329 
330  CReserveKey *keyChange = transaction.getPossibleKeyChange();
331  if(!wallet->CommitTransaction(*newTx, *keyChange, g_connman.get()))
332  return TransactionCommitFailed;
333 
334  CTransaction* t = (CTransaction*)newTx;
335  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
336  ssTx << *t;
337  transaction_array.append(&(ssTx[0]), ssTx.size());
338  }
339 
340  // Add addresses / update labels that we've sent to to the address book,
341  // and emit coinsSent signal for each recipient
342  Q_FOREACH(const SendCoinsRecipient &rcp, transaction.getRecipients())
343  {
344  // Don't touch the address book when we have a payment request
345  if (!rcp.paymentRequest.IsInitialized())
346  {
347  std::string strAddress = rcp.address.toStdString();
348  CTxDestination dest = CBitcoinAddress(strAddress).Get();
349  std::string strLabel = rcp.label.toStdString();
350  {
351  LOCK(wallet->cs_wallet);
352 
353  std::map<CTxDestination, CAddressBookData>::iterator mi = wallet->mapAddressBook.find(dest);
354 
355  // Check if we have a new address or an updated label
356  if (mi == wallet->mapAddressBook.end())
357  {
358  wallet->SetAddressBook(dest, strLabel, "send");
359  }
360  else if (mi->second.name != strLabel)
361  {
362  wallet->SetAddressBook(dest, strLabel, ""); // "" means don't change purpose
363  }
364  }
365  }
366  Q_EMIT coinsSent(wallet, rcp, transaction_array);
367  }
368  checkBalanceChanged(); // update balance immediately, otherwise there could be a short noticeable delay until pollBalanceChanged hits
369 
370  return SendCoinsReturn(OK);
371 }

586 // AvailableCoins + LockedCoins grouped by wallet address (put change in one group with wallet address)
587 void WalletModel::listCoins(std::map<QString, std::vector<COutput> >& mapCoins) const
588 {
589  std::vector<COutput> vCoins;
590  wallet->AvailableCoins(vCoins);
591 
592  LOCK2(cs_main, wallet->cs_wallet); // ListLockedCoins, mapWallet
593  std::vector<COutPoint> vLockedCoins;
594  wallet->ListLockedCoins(vLockedCoins);
595 
596  // add locked coins
597  BOOST_FOREACH(const COutPoint& outpoint, vLockedCoins)
598  {
599  if (!wallet->mapWallet.count(outpoint.hash)) continue;
600  int nDepth = wallet->mapWallet[outpoint.hash].GetDepthInMainChain();
601  if (nDepth < 0) continue;
602  COutput out(&wallet->mapWallet[outpoint.hash], outpoint.n, nDepth, true, true);
603  if (outpoint.n < out.tx->vout.size() && wallet->IsMine(out.tx->vout[outpoint.n]) == ISMINE_SPENDABLE)
604  vCoins.push_back(out);
605  }
606 
607  BOOST_FOREACH(const COutput& out, vCoins)
608  {
609  COutput cout = out;
610 
611  while (wallet->IsChange(cout.tx->vout[cout.i]) && cout.tx->vin.size() > 0 && wallet->IsMine(cout.tx->vin[0]))
612  {
613  if (!wallet->mapWallet.count(cout.tx->vin[0].prevout.hash)) break;
614  cout = COutput(&wallet->mapWallet[cout.tx->vin[0].prevout.hash], cout.tx->vin[0].prevout.n, 0, true, true);
615  }
616 
617  CTxDestination address;
618  if(!out.fSpendable || !ExtractDestination(cout.tx->vout[cout.i].scriptPubKey, address))
619  continue;
620  mapCoins[QString::fromStdString(CBitcoinAddress(address).ToString())].push_back(out);
621  }
622 }
