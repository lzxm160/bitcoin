105 //
106 // Thread management and startup/shutdown:
107 //
108 // The network-processing threads are all part of a thread group
109 // created by AppInit() or the Qt main() function.
110 //
111 // A clean exit happens when StartShutdown() or the SIGTERM
112 // signal handler sets fRequestShutdown, which triggers
113 // the DetectShutdownThread(), which interrupts the main thread group.
114 // DetectShutdownThread() then exits, which causes AppInit() to
115 // continue (it .joins the shutdown thread).
116 // Shutdown() is then
117 // called to clean up database connections, and stop other
118 // threads that should only be stopped after the main network-processing
119 // threads have exited.
120 //
121 // Note that if running -daemon the parent process returns from AppInit2
122 // before adding any threads to the threadGroup, so .join_all() returns
123 // immediately and the parent exits from main().
124 //
125 // Shutdown for Qt is very similar, only it uses a QTimer to detect
126 // fRequestShutdown getting set, and then does the normal Qt
127 // shutdown thing.
128 //

180 void Shutdown()
181 {
182  LogPrintf("%s: In progress...\n", __func__);
183  static CCriticalSection cs_Shutdown;
184  TRY_LOCK(cs_Shutdown, lockShutdown);
185  if (!lockShutdown)
186  return;
187 
192  RenameThread("bitcoin-shutoff");
193  mempool.AddTransactionsUpdated(1);
194 
195  StopHTTPRPC();
196  StopREST();
197  StopRPC();
198  StopHTTPServer();
199 #ifdef ENABLE_WALLET
200  if (pwalletMain)
201  pwalletMain->Flush(false);
202 #endif
203  MapPort(false);
204  UnregisterValidationInterface(peerLogic.get());
205  peerLogic.reset();
206  g_connman.reset();
207 
208  StopTorControl();
209  UnregisterNodeSignals(GetNodeSignals());
210 
211  if (fFeeEstimatesInitialized)
212  {
213  boost::filesystem::path est_path = GetDataDir() / FEE_ESTIMATES_FILENAME;
214  CAutoFile est_fileout(fopen(est_path.string().c_str(), "wb"), SER_DISK, CLIENT_VERSION);
215  if (!est_fileout.IsNull())
216  mempool.WriteFeeEstimates(est_fileout);
217  else
218  LogPrintf("%s: Failed to write fee estimates to %s\n", __func__, est_path.string());
219  fFeeEstimatesInitialized = false;
220  }
221 
222  {
223  LOCK(cs_main);
224  if (pcoinsTip != NULL) {
225  FlushStateToDisk();
226  }
227  delete pcoinsTip;
228  pcoinsTip = NULL;
229  delete pcoinscatcher;
230  pcoinscatcher = NULL;
231  delete pcoinsdbview;
232  pcoinsdbview = NULL;
233  delete pblocktree;
234  pblocktree = NULL;
235  }
236 #ifdef ENABLE_WALLET
237  if (pwalletMain)
238  pwalletMain->Flush(true);
239 #endif
240 
241 #if ENABLE_ZMQ
242  if (pzmqNotificationInterface) {
243  UnregisterValidationInterface(pzmqNotificationInterface);
244  delete pzmqNotificationInterface;
245  pzmqNotificationInterface = NULL;
246  }
247 #endif
248 
249 #ifndef WIN32
250  try {
251  boost::filesystem::remove(GetPidFile());
252  } catch (const boost::filesystem::filesystem_error& e) {
253  LogPrintf("%s: Unable to remove pidfile: %s\n", __func__, e.what());
254  }
255 #endif
256  UnregisterAllValidationInterfaces();
257 #ifdef ENABLE_WALLET
258  delete pwalletMain;
259  pwalletMain = NULL;
260 #endif
261  globalVerifyHandle.reset();
262  ECC_Stop();
263  LogPrintf("%s: done\n", __func__);
264 }

557 // If we're using -prune with -reindex, then delete block files that will be ignored by the
558 // reindex. Since reindexing works by starting at block file 0 and looping until a blockfile
559 // is missing, do the same here to delete any later block files after a gap. Also delete all
560 // rev files since they'll be rewritten by the reindex anyway. This ensures that vinfoBlockFile
561 // is in sync with what's actually on disk by the time we start downloading, so that pruning
562 // works correctly.
563 void CleanupBlockRevFiles()
564 {
565  using namespace boost::filesystem;
566  map<string, path> mapBlockFiles;
567 
568  // Glob all blk?????.dat and rev?????.dat files from the blocks directory.
569  // Remove the rev files immediately and insert the blk file paths into an
570  // ordered map keyed by block file index.
571  LogPrintf("Removing unusable blk?????.dat and rev?????.dat files for -reindex with -prune\n");
572  path blocksdir = GetDataDir() / "blocks";
573  for (directory_iterator it(blocksdir); it != directory_iterator(); it++) {
574  if (is_regular_file(*it) &&
575  it->path().filename().string().length() == 12 &&
576  it->path().filename().string().substr(8,4) == ".dat")
577  {
578  if (it->path().filename().string().substr(0,3) == "blk")
579  mapBlockFiles[it->path().filename().string().substr(3,5)] = it->path();
580  else if (it->path().filename().string().substr(0,3) == "rev")
581  remove(it->path());
582  }
583  }
584 
585  // Remove all block files that aren't part of a contiguous set starting at
586  // zero by walking the ordered map (keys are block file indices) by
587  // keeping a separate counter. Once we hit a gap (or if 0 doesn't exist)
588  // start removing block files.
589  int nContigCounter = 0;
590  BOOST_FOREACH(const PAIRTYPE(string, path)& item, mapBlockFiles) {
591  if (atoi(item.first) == nContigCounter) {
592  nContigCounter++;
593  continue;
594  }
595  remove(item.second);
596  }
597 }
