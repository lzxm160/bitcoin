21 UniValue CallRPC(string args)
22 {
23  vector<string> vArgs;
24  boost::split(vArgs, args, boost::is_any_of(" \t"));
25  string strMethod = vArgs[0];
26  vArgs.erase(vArgs.begin());
27  UniValue params = RPCConvertValues(strMethod, vArgs);
28  BOOST_CHECK(tableRPC[strMethod]);
29  rpcfn_type method = tableRPC[strMethod]->actor;
30  try {
31  UniValue result = (*method)(params, false);
32  return result;
33  }
34  catch (const UniValue& objError) {
35  throw runtime_error(find_value(objError, "message").get_str());
36  }
37 }
