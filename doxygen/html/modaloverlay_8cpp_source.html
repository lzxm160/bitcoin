75 void ModalOverlay::tipUpdate(int count, const QDateTime& blockDate, double nVerificationProgress)
76 {
77  QDateTime currentDate = QDateTime::currentDateTime();
78 
79  // keep a vector of samples of verification progress at height
80  blockProcessTime.push_front(qMakePair(currentDate.toMSecsSinceEpoch(), nVerificationProgress));
81 
82  // show progress speed if we have more then one sample
83  if (blockProcessTime.size() >= 2)
84  {
85  double progressStart = blockProcessTime[0].second;
86  double progressDelta = 0;
87  double progressPerHour = 0;
88  qint64 timeDelta = 0;
89  qint64 remainingMSecs = 0;
90  double remainingProgress = 1.0 - nVerificationProgress;
91  for (int i = 1; i < blockProcessTime.size(); i++)
92  {
93  QPair<qint64, double> sample = blockProcessTime[i];
94 
95  // take first sample after 500 seconds or last available one
96  if (sample.first < (currentDate.toMSecsSinceEpoch() - 500 * 1000) || i == blockProcessTime.size() - 1) {
97  progressDelta = progressStart-sample.second;
98  timeDelta = blockProcessTime[0].first - sample.first;
99  progressPerHour = progressDelta/(double)timeDelta*1000*3600;
100  remainingMSecs = remainingProgress / progressDelta * timeDelta;
101  break;
102  }
103  }
104  // show progress increase per hour
105  ui->progressIncreasePerH->setText(QString::number(progressPerHour*100, 'f', 2)+"%");
106 
107  // show expected remaining time
108  ui->expectedTimeLeft->setText(GUIUtil::formateNiceTimeOffset(remainingMSecs/1000.0));
109 
110  static const int MAX_SAMPLES = 5000;
111  if (blockProcessTime.count() > MAX_SAMPLES)
112  blockProcessTime.remove(MAX_SAMPLES, blockProcessTime.count()-MAX_SAMPLES);
113  }
114 
115  // show the last block date
116  ui->newestBlockDate->setText(blockDate.toString());
117 
118  // show the percentage done according to nVerificationProgress
119  ui->percentageProgress->setText(QString::number(nVerificationProgress*100, 'f', 2)+"%");
120  ui->progressBar->setValue(nVerificationProgress*100);
121 
122  if (!bestHeaderDate.isValid())
123  // not syncing
124  return;
125 
126  // estimate the number of headers left based on nPowTargetSpacing
127  // and check if the gui is not aware of the the best header (happens rarely)
128  int estimateNumHeadersLeft = bestHeaderDate.secsTo(currentDate) / 600;
129  bool hasBestHeader = bestHeaderHeight >= count;
130 
131  // show remaining number of blocks
132  if (estimateNumHeadersLeft < 24 && hasBestHeader) {
133  ui->numberOfBlocksLeft->setText(QString::number(bestHeaderHeight - count));
134  } else {
135  ui->expectedTimeLeft->setText(tr("Unknown. Syncing Headers..."));
136  }
137 }
