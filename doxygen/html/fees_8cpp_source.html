78 // returns -1 on error conditions
79 double TxConfirmStats::EstimateMedianVal(int confTarget, double sufficientTxVal,
80  double successBreakPoint, bool requireGreater,
81  unsigned int nBlockHeight)
82 {
83  // Counters for a bucket (or range of buckets)
84  double nConf = 0; // Number of tx's confirmed within the confTarget
85  double totalNum = 0; // Total number of tx's that were ever confirmed
86  int extraNum = 0; // Number of tx's still in mempool for confTarget or longer
87 
88  int maxbucketindex = buckets.size() - 1;
89 
90  // requireGreater means we are looking for the lowest fee/priority such that all higher
91  // values pass, so we start at maxbucketindex (highest fee) and look at successively
92  // smaller buckets until we reach failure. Otherwise, we are looking for the highest
93  // fee/priority such that all lower values fail, and we go in the opposite direction.
94  unsigned int startbucket = requireGreater ? maxbucketindex : 0;
95  int step = requireGreater ? -1 : 1;
96 
97  // We'll combine buckets until we have enough samples.
98  // The near and far variables will define the range we've combined
99  // The best variables are the last range we saw which still had a high
100  // enough confirmation rate to count as success.
101  // The cur variables are the current range we're counting.
102  unsigned int curNearBucket = startbucket;
103  unsigned int bestNearBucket = startbucket;
104  unsigned int curFarBucket = startbucket;
105  unsigned int bestFarBucket = startbucket;
106 
107  bool foundAnswer = false;
108  unsigned int bins = unconfTxs.size();
109 
110  // Start counting from highest(default) or lowest fee/pri transactions
111  for (int bucket = startbucket; bucket >= 0 && bucket <= maxbucketindex; bucket += step) {
112  curFarBucket = bucket;
113  nConf += confAvg[confTarget - 1][bucket];
114  totalNum += txCtAvg[bucket];
115  for (unsigned int confct = confTarget; confct < GetMaxConfirms(); confct++)
116  extraNum += unconfTxs[(nBlockHeight - confct)%bins][bucket];
117  extraNum += oldUnconfTxs[bucket];
118  // If we have enough transaction data points in this range of buckets,
119  // we can test for success
120  // (Only count the confirmed data points, so that each confirmation count
121  // will be looking at the same amount of data and same bucket breaks)
122  if (totalNum >= sufficientTxVal / (1 - decay)) {
123  double curPct = nConf / (totalNum + extraNum);
124 
125  // Check to see if we are no longer getting confirmed at the success rate
126  if (requireGreater && curPct < successBreakPoint)
127  break;
128  if (!requireGreater && curPct > successBreakPoint)
129  break;
130 
131  // Otherwise update the cumulative stats, and the bucket variables
132  // and reset the counters
133  else {
134  foundAnswer = true;
135  nConf = 0;
136  totalNum = 0;
137  extraNum = 0;
138  bestNearBucket = curNearBucket;
139  bestFarBucket = curFarBucket;
140  curNearBucket = bucket + step;
141  }
142  }
143  }
144 
145  double median = -1;
146  double txSum = 0;
147 
148  // Calculate the "average" fee of the best bucket range that met success conditions
149  // Find the bucket with the median transaction and then report the average fee from that bucket
150  // This is a compromise between finding the median which we can't since we don't save all tx's
151  // and reporting the average which is less accurate
152  unsigned int minBucket = bestNearBucket < bestFarBucket ? bestNearBucket : bestFarBucket;
153  unsigned int maxBucket = bestNearBucket > bestFarBucket ? bestNearBucket : bestFarBucket;
154  for (unsigned int j = minBucket; j <= maxBucket; j++) {
155  txSum += txCtAvg[j];
156  }
157  if (foundAnswer && txSum != 0) {
158  txSum = txSum / 2;
159  for (unsigned int j = minBucket; j <= maxBucket; j++) {
160  if (txCtAvg[j] < txSum)
161  txSum -= txCtAvg[j];
162  else { // we're in the right bucket
163  median = avg[j] / txCtAvg[j];
164  break;
165  }
166  }
167  }
168 
169  LogPrint("estimatefee", "%3d: For conf success %s %4.2f need %s %s: %12.5g from buckets %8g - %8g Cur Bucket stats %6.2f%% %8.1f/(%.1f+%d mempool)\n",
170  confTarget, requireGreater ? ">" : "<", successBreakPoint, dataTypeString,
171  requireGreater ? ">" : "<", median, buckets[minBucket], buckets[maxBucket],
172  100 * nConf / (totalNum + extraNum), nConf, totalNum, extraNum);
173 
174  return median;
175 }

186 void TxConfirmStats::Read(CAutoFile& filein)
187 {
188  // Read data file into temporary variables and do some very basic sanity checking
189  std::vector<double> fileBuckets;
190  std::vector<double> fileAvg;
191  std::vector<std::vector<double> > fileConfAvg;
192  std::vector<double> fileTxCtAvg;
193  double fileDecay;
194  size_t maxConfirms;
195  size_t numBuckets;
196 
197  filein >> fileDecay;
198  if (fileDecay <= 0 || fileDecay >= 1)
199  throw std::runtime_error("Corrupt estimates file. Decay must be between 0 and 1 (non-inclusive)");
200  filein >> fileBuckets;
201  numBuckets = fileBuckets.size();
202  if (numBuckets <= 1 || numBuckets > 1000)
203  throw std::runtime_error("Corrupt estimates file. Must have between 2 and 1000 fee/pri buckets");
204  filein >> fileAvg;
205  if (fileAvg.size() != numBuckets)
206  throw std::runtime_error("Corrupt estimates file. Mismatch in fee/pri average bucket count");
207  filein >> fileTxCtAvg;
208  if (fileTxCtAvg.size() != numBuckets)
209  throw std::runtime_error("Corrupt estimates file. Mismatch in tx count bucket count");
210  filein >> fileConfAvg;
211  maxConfirms = fileConfAvg.size();
212  if (maxConfirms <= 0 || maxConfirms > 6 * 24 * 7) // one week
213  throw std::runtime_error("Corrupt estimates file. Must maintain estimates for between 1 and 1008 (one week) confirms");
214  for (unsigned int i = 0; i < maxConfirms; i++) {
215  if (fileConfAvg[i].size() != numBuckets)
216  throw std::runtime_error("Corrupt estimates file. Mismatch in fee/pri conf average bucket count");
217  }
218  // Now that we've processed the entire fee estimate data file and not
219  // thrown any errors, we can copy it to our data structures
220  decay = fileDecay;
221  buckets = fileBuckets;
222  avg = fileAvg;
223  confAvg = fileConfAvg;
224  txCtAvg = fileTxCtAvg;
225  bucketMap.clear();
226 
227  // Resize the current block variables which aren't stored in the data file
228  // to match the number of confirms and buckets
229  curBlockConf.resize(maxConfirms);
230  for (unsigned int i = 0; i < maxConfirms; i++) {
231  curBlockConf[i].resize(buckets.size());
232  }
233  curBlockTxCt.resize(buckets.size());
234  curBlockVal.resize(buckets.size());
235 
236  unconfTxs.resize(maxConfirms);
237  for (unsigned int i = 0; i < maxConfirms; i++) {
238  unconfTxs[i].resize(buckets.size());
239  }
240  oldUnconfTxs.resize(buckets.size());
241 
242  for (unsigned int i = 0; i < buckets.size(); i++)
243  bucketMap[buckets[i]] = i;
244 
245  LogPrint("estimatefee", "Reading estimates: %u %s buckets counting confirms up to %u blocks\n",
246  numBuckets, dataTypeString, maxConfirms);
247 }

259 {
260  //nBestSeenHeight is not updated yet for the new block
261  int blocksAgo = nBestSeenHeight - entryHeight;
262  if (nBestSeenHeight == 0) // the BlockPolicyEstimator hasn't seen any blocks yet
263  blocksAgo = 0;

277  unsigned int blockIndex = entryHeight % unconfTxs.size();
278  if (unconfTxs[blockIndex][bucketindex] > 0)
279  unconfTxs[blockIndex][bucketindex]--;

281  LogPrint("estimatefee", "Blockpolicy error, mempool tx removed from blockIndex=%u,bucketIndex=%u already\n",
282  blockIndex, bucketindex);

346 void CBlockPolicyEstimator::processTransaction(const CTxMemPoolEntry& entry, bool fCurrentEstimate)
347 {
348  unsigned int txHeight = entry.GetHeight();
349  uint256 hash = entry.GetTx().GetHash();
350  if (mapMemPoolTxs[hash].stats != NULL) {
351  LogPrint("estimatefee", "Blockpolicy error mempool tx %s already being tracked\n",
352  hash.ToString().c_str());
353  return;
354  }
355 
356  if (txHeight < nBestSeenHeight) {
357  // Ignore side chains and re-orgs; assuming they are random they don't
358  // affect the estimate. We'll potentially double count transactions in 1-block reorgs.
359  return;
360  }
361 
362  // Only want to be updating estimates when our blockchain is synced,
363  // otherwise we'll miscalculate how many blocks its taking to get included.
364  if (!fCurrentEstimate)
365  return;
366 
367  if (!entry.WasClearAtEntry()) {
368  // This transaction depends on other transactions in the mempool to
369  // be included in a block before it will be able to be included, so
370  // we shouldn't include it in our calculations
371  return;
372  }
373 
374  // Fees are stored and reported as BTC-per-kb:
375  CFeeRate feeRate(entry.GetFee(), entry.GetTxSize());
376 
377  // Want the priority of the tx at confirmation. However we don't know
378  // what that will be and its too hard to continue updating it
379  // so use starting priority as a proxy
380  double curPri = entry.GetPriority(txHeight);
381  mapMemPoolTxs[hash].blockHeight = txHeight;
382 
383  LogPrint("estimatefee", "Blockpolicy mempool tx %s ", hash.ToString().substr(0,10));
384  // Record this as a priority estimate
385  if (entry.GetFee() == 0 || isPriDataPoint(feeRate, curPri)) {
386  mapMemPoolTxs[hash].stats = &priStats;
387  mapMemPoolTxs[hash].bucketIndex = priStats.NewTx(txHeight, curPri);
388  }
389  // Record this as a fee estimate
390  else if (isFeeDataPoint(feeRate, curPri)) {
391  mapMemPoolTxs[hash].stats = &feeStats;
392  mapMemPoolTxs[hash].bucketIndex = feeStats.NewTx(txHeight, (double)feeRate.GetFeePerK());
393  }
394  else {
395  LogPrint("estimatefee", "not adding");
396  }
397  LogPrint("estimatefee", "\n");
398 }

400 void CBlockPolicyEstimator::processBlockTx(unsigned int nBlockHeight, const CTxMemPoolEntry& entry)
401 {
402  if (!entry.WasClearAtEntry()) {
403  // This transaction depended on other transactions in the mempool to
404  // be included in a block before it was able to be included, so
405  // we shouldn't include it in our calculations
406  return;
407  }
408 
409  // How many blocks did it take for miners to include this transaction?
410  // blocksToConfirm is 1-based, so a transaction included in the earliest
411  // possible block has confirmation count of 1
412  int blocksToConfirm = nBlockHeight - entry.GetHeight();
413  if (blocksToConfirm <= 0) {
414  // This can't happen because we don't process transactions from a block with a height
415  // lower than our greatest seen height
416  LogPrint("estimatefee", "Blockpolicy error Transaction had negative blocksToConfirm\n");
417  return;
418  }
419 
420  // Fees are stored and reported as BTC-per-kb:
421  CFeeRate feeRate(entry.GetFee(), entry.GetTxSize());
422 
423  // Want the priority of the tx at confirmation. The priority when it
424  // entered the mempool could easily be very small and change quickly
425  double curPri = entry.GetPriority(nBlockHeight);
426 
427  // Record this as a priority estimate
428  if (entry.GetFee() == 0 || isPriDataPoint(feeRate, curPri)) {
429  priStats.Record(blocksToConfirm, curPri);
430  }
431  // Record this as a fee estimate
432  else if (isFeeDataPoint(feeRate, curPri)) {
433  feeStats.Record(blocksToConfirm, (double)feeRate.GetFeePerK());
434  }
435 }

437 void CBlockPolicyEstimator::processBlock(unsigned int nBlockHeight,
438  std::vector<CTxMemPoolEntry>& entries, bool fCurrentEstimate)
439 {
440  if (nBlockHeight <= nBestSeenHeight) {
441  // Ignore side chains and re-orgs; assuming they are random
442  // they don't affect the estimate.
443  // And if an attacker can re-org the chain at will, then
444  // you've got much bigger problems than "attacker can influence
445  // transaction fees."
446  return;
447  }
448  nBestSeenHeight = nBlockHeight;
449 
450  // Only want to be updating estimates when our blockchain is synced,
451  // otherwise we'll miscalculate how many blocks its taking to get included.
452  if (!fCurrentEstimate)
453  return;
454 
455  // Update the dynamic cutoffs
456  // a fee/priority is "likely" the reason your tx was included in a block if >85% of such tx's
457  // were confirmed in 2 blocks and is "unlikely" if <50% were confirmed in 10 blocks
458  LogPrint("estimatefee", "Blockpolicy recalculating dynamic cutoffs:\n");
459  priLikely = priStats.EstimateMedianVal(2, SUFFICIENT_PRITXS, MIN_SUCCESS_PCT, true, nBlockHeight);
460  if (priLikely == -1)
461  priLikely = INF_PRIORITY;
462 
463  double feeLikelyEst = feeStats.EstimateMedianVal(2, SUFFICIENT_FEETXS, MIN_SUCCESS_PCT, true, nBlockHeight);
464  if (feeLikelyEst == -1)
465  feeLikely = CFeeRate(INF_FEERATE);
466  else
467  feeLikely = CFeeRate(feeLikelyEst);
468 
469  priUnlikely = priStats.EstimateMedianVal(10, SUFFICIENT_PRITXS, UNLIKELY_PCT, false, nBlockHeight);
470  if (priUnlikely == -1)
471  priUnlikely = 0;
472 
473  double feeUnlikelyEst = feeStats.EstimateMedianVal(10, SUFFICIENT_FEETXS, UNLIKELY_PCT, false, nBlockHeight);
474  if (feeUnlikelyEst == -1)
475  feeUnlikely = CFeeRate(0);
476  else
477  feeUnlikely = CFeeRate(feeUnlikelyEst);
478 
479  // Clear the current block states
480  feeStats.ClearCurrent(nBlockHeight);
481  priStats.ClearCurrent(nBlockHeight);
482 
483  // Repopulate the current block states
484  for (unsigned int i = 0; i < entries.size(); i++)
485  processBlockTx(nBlockHeight, entries[i]);
486 
487  // Update all exponential averages with the current block states
488  feeStats.UpdateMovingAverages();
489  priStats.UpdateMovingAverages();
490 
491  LogPrint("estimatefee", "Blockpolicy after updating estimates for %u confirmed entries, new mempool map size %u\n",
492  entries.size(), mapMemPoolTxs.size());
493 }
