148 static void http_request_done(struct evhttp_request *req, void *ctx)
149 {
150  HTTPReply *reply = static_cast<HTTPReply*>(ctx);
151 
152  if (req == NULL) {
153  /* If req is NULL, it means an error occurred while connecting: the
154  * error code will have been passed to http_error_cb.
155  */
156  reply->status = 0;
157  return;
158  }
159 
160  reply->status = evhttp_request_get_response_code(req);
161 
162  struct evbuffer *buf = evhttp_request_get_input_buffer(req);
163  if (buf)
164  {
165  size_t size = evbuffer_get_length(buf);
166  const char *data = (const char*)evbuffer_pullup(buf, size);
167  if (data)
168  reply->body = std::string(data, size);
169  evbuffer_drain(buf, size);
170  }
171 }

181 UniValue CallRPC(const string& strMethod, const UniValue& params)
182 {
183  std::string host = GetArg("-rpcconnect", DEFAULT_RPCCONNECT);
184  int port = GetArg("-rpcport", BaseParams().RPCPort());
185 
186  // Create event base
187  struct event_base *base = event_base_new(); // TODO RAII
188  if (!base)
189  throw runtime_error("cannot create event_base");
190 
191  // Synchronously look up hostname
192  struct evhttp_connection *evcon = evhttp_connection_base_new(base, NULL, host.c_str(), port); // TODO RAII
193  if (evcon == NULL)
194  throw runtime_error("create connection failed");
195  evhttp_connection_set_timeout(evcon, GetArg("-rpcclienttimeout", DEFAULT_HTTP_CLIENT_TIMEOUT));
196 
197  HTTPReply response;
198  struct evhttp_request *req = evhttp_request_new(http_request_done, (void*)&response); // TODO RAII
199  if (req == NULL)
200  throw runtime_error("create http request failed");
201 #if LIBEVENT_VERSION_NUMBER >= 0x02010300
202  evhttp_request_set_error_cb(req, http_error_cb);
203 #endif
204 
205  // Get credentials
206  std::string strRPCUserColonPass;
207  if (mapArgs["-rpcpassword"] == "") {
208  // Try fall back to cookie-based authentication if no password is provided
209  if (!GetAuthCookie(&strRPCUserColonPass)) {
210  throw runtime_error(strprintf(
211  _("Could not locate RPC credentials. No authentication cookie could be found, and no rpcpassword is set in the configuration file (%s)"),
212  GetConfigFile(GetArg("-conf", BITCOIN_CONF_FILENAME)).string().c_str()));
213 
214  }
215  } else {
216  strRPCUserColonPass = mapArgs["-rpcuser"] + ":" + mapArgs["-rpcpassword"];
217  }
218 
219  struct evkeyvalq *output_headers = evhttp_request_get_output_headers(req);
220  assert(output_headers);
221  evhttp_add_header(output_headers, "Host", host.c_str());
222  evhttp_add_header(output_headers, "Connection", "close");
223  evhttp_add_header(output_headers, "Authorization", (std::string("Basic ") + EncodeBase64(strRPCUserColonPass)).c_str());
224 
225  // Attach request data
226  std::string strRequest = JSONRPCRequest(strMethod, params, 1);
227  struct evbuffer * output_buffer = evhttp_request_get_output_buffer(req);
228  assert(output_buffer);
229  evbuffer_add(output_buffer, strRequest.data(), strRequest.size());
230 
231  int r = evhttp_make_request(evcon, req, EVHTTP_REQ_POST, "/");
232  if (r != 0) {
233  evhttp_connection_free(evcon);
234  event_base_free(base);
235  throw CConnectionFailed("send http request failed");
236  }
237 
238  event_base_dispatch(base);
239  evhttp_connection_free(evcon);
240  event_base_free(base);
241 
242  if (response.status == 0)
243  throw CConnectionFailed(strprintf("couldn't connect to server (%d %s)", response.error, http_errorstring(response.error)));
244  else if (response.status == HTTP_UNAUTHORIZED)
245  throw runtime_error("incorrect rpcuser or rpcpassword (authorization failed)");
246  else if (response.status >= 400 && response.status != HTTP_BAD_REQUEST && response.status != HTTP_NOT_FOUND && response.status != HTTP_INTERNAL_SERVER_ERROR)
247  throw runtime_error(strprintf("server returned HTTP error %d", response.status));
248  else if (response.body.empty())
249  throw runtime_error("no response from server");
250 
251  // Parse reply
252  UniValue valReply(UniValue::VSTR);
253  if (!valReply.read(response.body))
254  throw runtime_error("couldn't parse reply from server");
255  const UniValue& reply = valReply.get_obj();
256  if (reply.empty())
257  throw runtime_error("expected reply to have result, error and id properties");
258 
259  return reply;
260 }

262 int CommandLineRPC(int argc, char *argv[])
263 {
264  string strPrint;
265  int nRet = 0;
266  try {
267  // Skip switches
268  while (argc > 1 && IsSwitchChar(argv[1][0])) {
269  argc--;
270  argv++;
271  }
272  std::vector<std::string> args = std::vector<std::string>(&argv[1], &argv[argc]);
273  if (GetBoolArg("-stdin", false)) {
274  // Read one arg per line from stdin and append
275  std::string line;
276  while (std::getline(std::cin,line))
277  args.push_back(line);
278  }
279  if (args.size() < 1)
280  throw runtime_error("too few parameters (need at least command)");
281  std::string strMethod = args[0];
282  UniValue params = RPCConvertValues(strMethod, std::vector<std::string>(args.begin()+1, args.end()));
283 
284  // Execute and handle connection failures with -rpcwait
285  const bool fWait = GetBoolArg("-rpcwait", false);
286  do {
287  try {
288  const UniValue reply = CallRPC(strMethod, params);
289 
290  // Parse reply
291  const UniValue& result = find_value(reply, "result");
292  const UniValue& error = find_value(reply, "error");
293 
294  if (!error.isNull()) {
295  // Error
296  int code = error["code"].get_int();
297  if (fWait && code == RPC_IN_WARMUP)
298  throw CConnectionFailed("server in warmup");
299  strPrint = "error: " + error.write();
300  nRet = abs(code);
301  if (error.isObject())
302  {
303  UniValue errCode = find_value(error, "code");
304  UniValue errMsg = find_value(error, "message");
305  strPrint = errCode.isNull() ? "" : "error code: "+errCode.getValStr()+"\n";
306 
307  if (errMsg.isStr())
308  strPrint += "error message:\n"+errMsg.get_str();
309  }
310  } else {
311  // Result
312  if (result.isNull())
313  strPrint = "";
314  else if (result.isStr())
315  strPrint = result.get_str();
316  else
317  strPrint = result.write(2);
318  }
319  // Connection succeeded, no need to retry.
320  break;
321  }
322  catch (const CConnectionFailed&) {
323  if (fWait)
324  MilliSleep(1000);
325  else
326  throw;
327  }
328  } while (fWait);
329  }
330  catch (const boost::thread_interrupted&) {
331  throw;
332  }
333  catch (const std::exception& e) {
334  strPrint = string("error: ") + e.what();
335  nRet = EXIT_FAILURE;
336  }
337  catch (...) {
338  PrintExceptionContinue(NULL, "CommandLineRPC()");
339  throw;
340  }
341 
342  if (strPrint != "") {
343  fprintf((nRet == 0 ? stdout : stderr), "%s\n", strPrint.c_str());
344  }
345  return nRet;
346 }
