70 void RPCTypeCheck(const UniValue& params,
71  const list<UniValue::VType>& typesExpected,
72  bool fAllowNull)
73 {
74  unsigned int i = 0;
75  BOOST_FOREACH(UniValue::VType t, typesExpected)
76  {
77  if (params.size() <= i)
78  break;
79 
80  const UniValue& v = params[i];
81  if (!((v.type() == t) || (fAllowNull && (v.isNull()))))
82  {
83  string err = strprintf("Expected type %s, got %s",
84  uvTypeName(t), uvTypeName(v.type()));
85  throw JSONRPCError(RPC_TYPE_ERROR, err);
86  }
87  i++;
88  }
89 }

91 void RPCTypeCheckObj(const UniValue& o,
92  const map<string, UniValueType>& typesExpected,
93  bool fAllowNull,
94  bool fStrict)
95 {
96  for (const auto& t : typesExpected) {
97  const UniValue& v = find_value(o, t.first);
98  if (!fAllowNull && v.isNull())
99  throw JSONRPCError(RPC_TYPE_ERROR, strprintf("Missing %s", t.first));
100 
101  if (!(t.second.typeAny || v.type() == t.second.type || (fAllowNull && v.isNull()))) {
102  string err = strprintf("Expected type %s for %s, got %s",
103  uvTypeName(t.second.type), t.first, uvTypeName(v.type()));
104  throw JSONRPCError(RPC_TYPE_ERROR, err);
105  }
106  }
107 
108  if (fStrict)
109  {
110  BOOST_FOREACH(const string& k, o.getKeys())
111  {
112  if (typesExpected.count(k) == 0)
113  {
114  string err = strprintf("Unexpected key %s", k);
115  throw JSONRPCError(RPC_TYPE_ERROR, err);
116  }
117  }
118  }

176 std::string CRPCTable::help(const std::string& strCommand) const
177 {
178  string strRet;
179  string category;
180  set<rpcfn_type> setDone;
181  vector<pair<string, const CRPCCommand*> > vCommands;
182 
183  for (map<string, const CRPCCommand*>::const_iterator mi = mapCommands.begin(); mi != mapCommands.end(); ++mi)
184  vCommands.push_back(make_pair(mi->second->category + mi->first, mi->second));
185  sort(vCommands.begin(), vCommands.end());
186 
187  BOOST_FOREACH(const PAIRTYPE(string, const CRPCCommand*)& command, vCommands)
188  {
189  const CRPCCommand *pcmd = command.second;
190  string strMethod = pcmd->name;
191  // We already filter duplicates, but these deprecated screw up the sort order
192  if (strMethod.find("label") != string::npos)
193  continue;
194  if ((strCommand != "" || pcmd->category == "hidden") && strMethod != strCommand)
195  continue;
196  try
197  {
198  UniValue params;
199  rpcfn_type pfn = pcmd->actor;
200  if (setDone.insert(pfn).second)
201  (*pfn)(params, true);
202  }
203  catch (const std::exception& e)
204  {
205  // Help text is returned in an exception
206  string strHelp = string(e.what());
207  if (strCommand == "")
208  {
209  if (strHelp.find('\n') != string::npos)
210  strHelp = strHelp.substr(0, strHelp.find('\n'));
211 
212  if (category != pcmd->category)
213  {
214  if (!category.empty())
215  strRet += "\n";
216  category = pcmd->category;
217  string firstLetter = category.substr(0,1);
218  boost::to_upper(firstLetter);
219  strRet += "== " + firstLetter + category.substr(1) + " ==\n";
220  }
221  }
222  strRet += strHelp + "\n";
223  }
224  }
225  if (strRet == "")
226  strRet = strprintf("help: unknown command: %s\n", strCommand);
227  strRet = strRet.substr(0,strRet.size()-1);
228  return strRet;
229 }

357 void JSONRequest::parse(const UniValue& valRequest)
358 {
359  // Parse request
360  if (!valRequest.isObject())
361  throw JSONRPCError(RPC_INVALID_REQUEST, "Invalid Request object");
362  const UniValue& request = valRequest.get_obj();
363 
364  // Parse id now so errors from here on will have the id
365  id = find_value(request, "id");
366 
367  // Parse method
368  UniValue valMethod = find_value(request, "method");
369  if (valMethod.isNull())
370  throw JSONRPCError(RPC_INVALID_REQUEST, "Missing method");
371  if (!valMethod.isStr())
372  throw JSONRPCError(RPC_INVALID_REQUEST, "Method must be a string");
373  strMethod = valMethod.get_str();
374  if (strMethod != "getblocktemplate")
375  LogPrint("rpc", "ThreadRPCServer method=%s\n", SanitizeString(strMethod));
376 
377  // Parse params
378  UniValue valParams = find_value(request, "params");
379  if (valParams.isArray())
380  params = valParams.get_array();
381  else if (valParams.isNull())
382  params = UniValue(UniValue::VARR);
383  else
384  throw JSONRPCError(RPC_INVALID_REQUEST, "Params must be an array");
385 }

387 static UniValue JSONRPCExecOne(const UniValue& req)
388 {
389  UniValue rpc_result(UniValue::VOBJ);
390 
391  JSONRequest jreq;
392  try {
393  jreq.parse(req);
394 
395  UniValue result = tableRPC.execute(jreq.strMethod, jreq.params);
396  rpc_result = JSONRPCReplyObj(result, NullUniValue, jreq.id);
397  }
398  catch (const UniValue& objError)
399  {
400  rpc_result = JSONRPCReplyObj(NullUniValue, objError, jreq.id);
401  }
402  catch (const std::exception& e)
403  {
404  rpc_result = JSONRPCReplyObj(NullUniValue,
405  JSONRPCError(RPC_PARSE_ERROR, e.what()), jreq.id);
406  }
407 
408  return rpc_result;
409 }

420 UniValue CRPCTable::execute(const std::string &strMethod, const UniValue &params) const
421 {
422  // Return immediately if in warmup
423  {
424  LOCK(cs_rpcWarmup);
425  if (fRPCInWarmup)
426  throw JSONRPCError(RPC_IN_WARMUP, rpcWarmupStatus);
427  }
428 
429  // Find method
430  const CRPCCommand *pcmd = tableRPC[strMethod];
431  if (!pcmd)
432  throw JSONRPCError(RPC_METHOD_NOT_FOUND, "Method not found");
433 
434  g_rpcSignals.PreCommand(*pcmd);
435 
436  try
437  {
438  // Execute
439  return pcmd->actor(params, false);
440  }
441  catch (const std::exception& e)
442  {
443  throw JSONRPCError(RPC_MISC_ERROR, e.what());
444  }
445 
446  g_rpcSignals.PostCommand(*pcmd);
447 }
