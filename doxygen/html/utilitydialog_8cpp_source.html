174 void ShutdownWindow::showShutdownWindow(BitcoinGUI *window)
175 {
176  if (!window)
177  return;
178 
179  // Show a simple window indicating shutdown status
180  QWidget *shutdownWindow = new ShutdownWindow();
181  // We don't hold a direct pointer to the shutdown window after creation, so use
182  // Qt::WA_DeleteOnClose to make sure that the window will be deleted eventually.
183  shutdownWindow->setAttribute(Qt::WA_DeleteOnClose);
184  shutdownWindow->setWindowTitle(window->windowTitle());
185 
186  // Center shutdown window at where main window was
187  const QPoint global = window->mapToGlobal(window->rect().center());
188  shutdownWindow->move(global.x() - shutdownWindow->width() / 2, global.y() - shutdownWindow->height() / 2);
189  shutdownWindow->show();
190 }
