203 static bool isCanonicalException(const std::ios_base::failure& ex)
204 {
205  std::ios_base::failure expectedException("non-canonical ReadCompactSize()");
206 
207  // The string returned by what() can be different for different platforms.

272  BOOST_CHECK_EQUAL(ss[5], c);
273

275  BOOST_CHECK_EQUAL(ss.size(), 7);
276  BOOST_CHECK_EQUAL(ss[2], c);
277 
278  // Delete at beginning/end/middle

282
