124 void CWallet::DeriveNewChildKey(CKeyMetadata& metadata, CKey& secret)
125 {
126  // for now we use a fixed keypath scheme of m/0'/0'/k
127  CKey key; //master key seed (256bit)
128  CExtKey masterKey; //hd master key
129  CExtKey accountKey; //key at m/0'
130  CExtKey externalChainChildKey; //key at m/0'/0'
131  CExtKey childKey; //key at m/0'/0'/<n>'
132 
133  // try to get the master key
134  if (!GetKey(hdChain.masterKeyID, key))
135  throw std::runtime_error(std::string(__func__) + ": Master key not found");
136 
137  masterKey.SetMaster(key.begin(), key.size());
138 
139  // derive m/0'
140  // use hardened derivation (child keys >= 0x80000000 are hardened after bip32)
141  masterKey.Derive(accountKey, BIP32_HARDENED_KEY_LIMIT);
142 
143  // derive m/0'/0'
144  accountKey.Derive(externalChainChildKey, BIP32_HARDENED_KEY_LIMIT);
145 
146  // derive child key at next index, skip keys already known to the wallet
147  do {
148  // always derive hardened keys
149  // childIndex | BIP32_HARDENED_KEY_LIMIT = derive childIndex in hardened child-index-range
150  // example: 1 | BIP32_HARDENED_KEY_LIMIT == 0x80000001 == 2147483649
151  externalChainChildKey.Derive(childKey, hdChain.nExternalChainCounter | BIP32_HARDENED_KEY_LIMIT);
152  metadata.hdKeypath = "m/0'/0'/" + std::to_string(hdChain.nExternalChainCounter) + "'";
153  metadata.hdMasterKeyID = hdChain.masterKeyID;
154  // increment childkey index
155  hdChain.nExternalChainCounter++;
156  } while (HaveKey(childKey.key.GetPubKey().GetID()));
157  secret = childKey.key;
158 
159  // update the chain model in the database
160  if (!CWalletDB(strWalletFile).WriteHDChain(hdChain))
161  throw std::runtime_error(std::string(__func__) + ": Writing HD chain model failed");
162 }

758 bool CWallet::AddToWallet(const CWalletTx& wtxIn, bool fFlushOnClose)
759 {
760  LOCK(cs_wallet);
761 
762  CWalletDB walletdb(strWalletFile, "r+", fFlushOnClose);
763 
764  uint256 hash = wtxIn.GetHash();
765 
766  // Inserts only if not already there, returns tx inserted or tx found
767  pair<map<uint256, CWalletTx>::iterator, bool> ret = mapWallet.insert(make_pair(hash, wtxIn));
768  CWalletTx& wtx = (*ret.first).second;
769  wtx.BindWallet(this);
770  bool fInsertedNew = ret.second;
771  if (fInsertedNew)
772  {
773  wtx.nTimeReceived = GetAdjustedTime();
774  wtx.nOrderPos = IncOrderPosNext(&walletdb);
775  wtxOrdered.insert(make_pair(wtx.nOrderPos, TxPair(&wtx, (CAccountingEntry*)0)));
776 
777  wtx.nTimeSmart = wtx.nTimeReceived;
778  if (!wtxIn.hashUnset())
779  {
780  if (mapBlockIndex.count(wtxIn.hashBlock))
781  {
782  int64_t latestNow = wtx.nTimeReceived;
783  int64_t latestEntry = 0;
784  {
785  // Tolerate times up to the last timestamp in the wallet not more than 5 minutes into the future
786  int64_t latestTolerated = latestNow + 300;
787  const TxItems & txOrdered = wtxOrdered;
788  for (TxItems::const_reverse_iterator it = txOrdered.rbegin(); it != txOrdered.rend(); ++it)
789  {
790  CWalletTx *const pwtx = (*it).second.first;
791  if (pwtx == &wtx)
792  continue;
793  CAccountingEntry *const pacentry = (*it).second.second;
794  int64_t nSmartTime;
795  if (pwtx)
796  {
797  nSmartTime = pwtx->nTimeSmart;
798  if (!nSmartTime)
799  nSmartTime = pwtx->nTimeReceived;
800  }
801  else
802  nSmartTime = pacentry->nTime;
803  if (nSmartTime <= latestTolerated)
804  {
805  latestEntry = nSmartTime;
806  if (nSmartTime > latestNow)
807  latestNow = nSmartTime;
808  break;
809  }
810  }
811  }
812 
813  int64_t blocktime = mapBlockIndex[wtxIn.hashBlock]->GetBlockTime();
814  wtx.nTimeSmart = std::max(latestEntry, std::min(blocktime, latestNow));
815  }
816  else
817  LogPrintf("AddToWallet(): found %s in block %s not in index\n",
818  wtxIn.GetHash().ToString(),
819  wtxIn.hashBlock.ToString());
820  }
821  AddToSpends(hash);
822  }
823 
824  bool fUpdated = false;
825  if (!fInsertedNew)
826  {
827  // Merge
828  if (!wtxIn.hashUnset() && wtxIn.hashBlock != wtx.hashBlock)
829  {
830  wtx.hashBlock = wtxIn.hashBlock;
831  fUpdated = true;
832  }
833  // If no longer abandoned, update
834  if (wtxIn.hashBlock.IsNull() && wtx.isAbandoned())
835  {
836  wtx.hashBlock = wtxIn.hashBlock;
837  fUpdated = true;
838  }
839  if (wtxIn.nIndex != -1 && (wtxIn.nIndex != wtx.nIndex))
840  {
841  wtx.nIndex = wtxIn.nIndex;
842  fUpdated = true;
843  }
844  if (wtxIn.fFromMe && wtxIn.fFromMe != wtx.fFromMe)
845  {
846  wtx.fFromMe = wtxIn.fFromMe;
847  fUpdated = true;
848  }
849  }
850 
852  LogPrintf("AddToWallet %s %s%s\n", wtxIn.GetHash().ToString(), (fInsertedNew ? "new" : ""), (fUpdated ? "update" : ""));
853 
854  // Write to disk
855  if (fInsertedNew || fUpdated)
856  if (!walletdb.WriteTx(wtx))
857  return false;
858 
859  // Break debit/credit balance caches:
860  wtx.MarkDirty();
861 
862  // Notify UI of new or updated transaction
863  NotifyTransactionChanged(this, hash, fInsertedNew ? CT_NEW : CT_UPDATED);
864 
865  // notify an external script when a wallet transaction comes in or is updated
866  std::string strCmd = GetArg("-walletnotify", "");
867 
868  if ( !strCmd.empty())
869  {
870  boost::replace_all(strCmd, "%s", wtxIn.GetHash().GetHex());
871  boost::thread t(runCommand, strCmd); // thread runs free
872  }
873 
874  return true;
875 }

937 bool CWallet::AbandonTransaction(const uint256& hashTx)
938 {
939  LOCK2(cs_main, cs_wallet);
940 
941  // Do not flush the wallet here for performance reasons
942  CWalletDB walletdb(strWalletFile, "r+", false);
943 
944  std::set<uint256> todo;
945  std::set<uint256> done;
946 
947  // Can't mark abandoned if confirmed or in mempool
948  assert(mapWallet.count(hashTx));
949  CWalletTx& origtx = mapWallet[hashTx];
950  if (origtx.GetDepthInMainChain() > 0 || origtx.InMempool()) {
951  return false;
952  }
953 
954  todo.insert(hashTx);
955 
956  while (!todo.empty()) {
957  uint256 now = *todo.begin();
958  todo.erase(now);
959  done.insert(now);
960  assert(mapWallet.count(now));
961  CWalletTx& wtx = mapWallet[now];
962  int currentconfirm = wtx.GetDepthInMainChain();
963  // If the orig tx was not in block, none of its spends can be
964  assert(currentconfirm <= 0);
965  // if (currentconfirm < 0) {Tx and spends are already conflicted, no need to abandon}
966  if (currentconfirm == 0 && !wtx.isAbandoned()) {
967  // If the orig tx was not in block/mempool, none of its spends can be in mempool
968  assert(!wtx.InMempool());
969  wtx.nIndex = -1;
970  wtx.setAbandoned();
971  wtx.MarkDirty();
972  walletdb.WriteTx(wtx);
973  NotifyTransactionChanged(this, wtx.GetHash(), CT_UPDATED);
974  // Iterate over all its outputs, and mark transactions in the wallet that spend them abandoned too
975  TxSpends::const_iterator iter = mapTxSpends.lower_bound(COutPoint(hashTx, 0));
976  while (iter != mapTxSpends.end() && iter->first.hash == now) {
977  if (!done.count(iter->second)) {
978  todo.insert(iter->second);
979  }
980  iter++;
981  }
982  // If a transaction changes 'conflicted' state, that changes the balance
983  // available of the outputs it spends. So force those to be recomputed
984  BOOST_FOREACH(const CTxIn& txin, wtx.vin)
985  {
986  if (mapWallet.count(txin.prevout.hash))
987  mapWallet[txin.prevout.hash].MarkDirty();
988  }
989  }
990  }
991 
992  return true;

996 {

998 
999  int conflictconfirms = 0;
1000  if (mapBlockIndex.count(hashBlock)) {
1001  CBlockIndex* pindex = mapBlockIndex[hashBlock];
1002  if (chainActive.Contains(pindex)) {
1003  conflictconfirms = -(chainActive.Height() - pindex->nHeight + 1);
1004  }
1005  }
1006  // If number of conflict confirms cannot be determined, this means
1007  // that the block is still unknown or not yet part of the main chain,
1008  // for example when loading the wallet during a reindex. Do nothing in that
1009  // case.
1010  if (conflictconfirms >= 0)
1011  return;
1012 
1013  // Do not flush the wallet here for performance reasons
1014  CWalletDB walletdb(strWalletFile, "r+", false);
1015 
1016  std::set<uint256> todo;
1017  std::set<uint256> done;

1019  todo.insert(hashTx);
1020 
1021  while (!todo.empty()) {
1022  uint256 now = *todo.begin();
1023  todo.erase(now);
1024  done.insert(now);
1025  assert(mapWallet.count(now));
1026  CWalletTx& wtx = mapWallet[now];
1027  int currentconfirm = wtx.GetDepthInMainChain();
1028  if (conflictconfirms < currentconfirm) {
1029  // Block is 'more conflicted' than current confirm; update.
1030  // Mark transaction as conflicted with this block.
1031  wtx.nIndex = -1;
1032  wtx.hashBlock = hashBlock;
1033  wtx.MarkDirty();

1041  iter++;
