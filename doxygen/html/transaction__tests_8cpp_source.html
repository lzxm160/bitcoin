61 unsigned int ParseScriptFlags(string strFlags)
62 {
63  if (strFlags.empty()) {
64  return 0;
65  }
66  unsigned int flags = 0;
67  vector<string> words;
68  boost::algorithm::split(words, strFlags, boost::algorithm::is_any_of(","));
69 
70  BOOST_FOREACH(string word, words)
71  {
72  if (!mapFlagNames.count(word))
73  BOOST_ERROR("Bad test: unknown verification flag '" << word << "'");
74  flags |= mapFlagNames[word];
75  }
76 
77  return flags;
78 }
