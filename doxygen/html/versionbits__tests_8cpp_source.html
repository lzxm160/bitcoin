189  // Sanity checks of version bit deployments
190  const Consensus::Params &mainnetParams = Params(CBaseChainParams::MAIN).GetConsensus();
191  for (int i=0; i<(int) Consensus::MAX_VERSION_BITS_DEPLOYMENTS; i++) {
192  uint32_t bitmask = VersionBitsMask(mainnetParams, (Consensus::DeploymentPos)i);
193  // Make sure that no deployment tries to set an invalid bit.
194  BOOST_CHECK_EQUAL(bitmask & ~(uint32_t)VERSIONBITS_TOP_MASK, bitmask);
195 
196  // Verify that the deployment windows of different deployment using the
197  // same bit are disjoint.
198  // This test may need modification at such time as a new deployment
199  // is proposed that reuses the bit of an activated soft fork, before the
200  // end time of that soft fork. (Alternatively, the end time of that
201  // activated soft fork could be later changed to be earlier to avoid
202  // overlap.)
203  for (int j=i+1; j<(int) Consensus::MAX_VERSION_BITS_DEPLOYMENTS; j++) {
204  if (VersionBitsMask(mainnetParams, (Consensus::DeploymentPos)j) == bitmask) {
205  BOOST_CHECK(mainnetParams.vDeployments[j].nStartTime > mainnetParams.vDeployments[i].nTimeout ||
206  mainnetParams.vDeployments[i].nStartTime > mainnetParams.vDeployments[j].nTimeout);
207  }
208  }

212 BOOST_AUTO_TEST_CASE(versionbits_computeblockversion)
213 {
214  // Check that ComputeBlockVersion will set the appropriate bit correctly
215  // on mainnet.
216  const Consensus::Params &mainnetParams = Params(CBaseChainParams::MAIN).GetConsensus();
217 
218  // Use the TESTDUMMY deployment for testing purposes.
219  int64_t bit = mainnetParams.vDeployments[Consensus::DEPLOYMENT_TESTDUMMY].bit;
220  int64_t nStartTime = mainnetParams.vDeployments[Consensus::DEPLOYMENT_TESTDUMMY].nStartTime;
221  int64_t nTimeout = mainnetParams.vDeployments[Consensus::DEPLOYMENT_TESTDUMMY].nTimeout;
222 
223  assert(nStartTime < nTimeout);
224 
225  // In the first chain, test that the bit is set by CBV until it has failed.
226  // In the second chain, test the bit is set by CBV while STARTED and
227  // LOCKED-IN, and then no longer set while ACTIVE.
228  VersionBitsTester firstChain, secondChain;
229 
230  // Start generating blocks before nStartTime
231  int64_t nTime = nStartTime - 1;
232 
233  // Before MedianTimePast of the chain has crossed nStartTime, the bit
234  // should not be set.
235  CBlockIndex *lastBlock = NULL;
236  lastBlock = firstChain.Mine(2016, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
237  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit), 0);
238 
239  // Mine 2011 more blocks at the old time, and check that CBV isn't setting the bit yet.
240  for (int i=1; i<2012; i++) {
241  lastBlock = firstChain.Mine(2016+i, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
242  // This works because VERSIONBITS_LAST_OLD_BLOCK_VERSION happens
243  // to be 4, and the bit we're testing happens to be bit 28.
244  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit), 0);
245  }
246  // Now mine 5 more blocks at the start time -- MTP should not have passed yet, so
247  // CBV should still not yet set the bit.
248  nTime = nStartTime;
249  for (int i=2012; i<=2016; i++) {
250  lastBlock = firstChain.Mine(2016+i, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
251  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit), 0);
252  }
253 
254  // Advance to the next period and transition to STARTED,
255  lastBlock = firstChain.Mine(6048, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
256  // so ComputeBlockVersion should now set the bit,
257  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
258  // and should also be using the VERSIONBITS_TOP_BITS.
259  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & VERSIONBITS_TOP_MASK, VERSIONBITS_TOP_BITS);
260 
261  // Check that ComputeBlockVersion will set the bit until nTimeout
262  nTime += 600;
263  int blocksToMine = 4032; // test blocks for up to 2 time periods
264  int nHeight = 6048;
265  // These blocks are all before nTimeout is reached.
266  while (nTime < nTimeout && blocksToMine > 0) {
267  lastBlock = firstChain.Mine(nHeight+1, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
268  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
269  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & VERSIONBITS_TOP_MASK, VERSIONBITS_TOP_BITS);
270  blocksToMine--;
271  nTime += 600;
272  nHeight += 1;
273  };
274 
275  nTime = nTimeout;
276  // FAILED is only triggered at the end of a period, so CBV should be setting
277  // the bit until the period transition.
278  for (int i=0; i<2015; i++) {
279  lastBlock = firstChain.Mine(nHeight+1, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
280  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
281  nHeight += 1;
282  }
283  // The next block should trigger no longer setting the bit.
284  lastBlock = firstChain.Mine(nHeight+1, nTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
285  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit), 0);
286 
287  // On a new chain:
288  // verify that the bit will be set after lock-in, and then stop being set
289  // after activation.
290  nTime = nStartTime;
291 
292  // Mine one period worth of blocks, and check that the bit will be on for the
293  // next period.
294  lastBlock = secondChain.Mine(2016, nStartTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
295  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
296 
297  // Mine another period worth of blocks, signaling the new bit.
298  lastBlock = secondChain.Mine(4032, nStartTime, VERSIONBITS_TOP_BITS | (1<<bit)).Tip();
299  // After one period of setting the bit on each block, it should have locked in.
300  // We keep setting the bit for one more period though, until activation.
301  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
302 
303  // Now check that we keep mining the block until the end of this period, and
304  // then stop at the beginning of the next period.
305  lastBlock = secondChain.Mine(6047, nStartTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
306  BOOST_CHECK((ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit)) != 0);
307  lastBlock = secondChain.Mine(6048, nStartTime, VERSIONBITS_LAST_OLD_BLOCK_VERSION).Tip();
308  BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & (1<<bit), 0);
309 
310  // Finally, verify that after a soft fork has activated, CBV no longer uses
311  // VERSIONBITS_LAST_OLD_BLOCK_VERSION.
312  //BOOST_CHECK_EQUAL(ComputeBlockVersion(lastBlock, mainnetParams) & VERSIONBITS_TOP_MASK, VERSIONBITS_TOP_BITS);
313 }
