148  // Must be valid private key

154  BOOST_CHECK_MESSAGE(privkey.size() == exp_payload.size() && std::equal(privkey.begin(), privkey.end(), exp_payload.begin()), "key mismatch:" + strTest);
155 
156  // Private key must be invalid public key

158  BOOST_CHECK_MESSAGE(!addr.IsValid(), "IsValid privkey as pubkey:" + strTest);
159  }
160  else
161  {
162  std::string exp_addrType = find_value(metadata, "addrType").get_str(); // "script" or "pubkey"
163  // Must be valid public key

165  BOOST_CHECK_MESSAGE(addr.IsValid(), "!IsValid:" + strTest);
166  BOOST_CHECK_MESSAGE(addr.IsScript() == (exp_addrType == "script"), "isScript mismatch" + strTest);
167  CTxDestination dest = addr.Get();
168  BOOST_CHECK_MESSAGE(boost::apply_visitor(TestAddrTypeVisitor(exp_addrType), dest), "addrType mismatch" + strTest);
169 
170  // Public key must be invalid private key

172  BOOST_CHECK_MESSAGE(!secret.IsValid(), "IsValid pubkey as privkey:" + strTest);
