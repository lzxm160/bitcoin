91 // This is a large randomized insert/remove simulation test on a variable-size
92 // stack of caches on top of CCoinsViewTest.
93 //
94 // It will randomly create/update/delete CCoins entries to a tip of caches, with
95 // txids picked from a limited list of random 256-bit hashes. Occasionally, a
96 // new tip is added to the stack of caches, or the tip is flushed and removed.
97 //
98 // During the process, booleans are kept to make sure that the randomized
99 // operation hits all branches.
100 BOOST_AUTO_TEST_CASE(coins_cache_simulation_test)
101 {
102  // Various coverage trackers.
103  bool removed_all_caches = false;
104  bool reached_4_caches = false;
105  bool added_an_entry = false;
106  bool removed_an_entry = false;
107  bool updated_an_entry = false;
108  bool found_an_entry = false;
109  bool missed_an_entry = false;
110 
111  // A simple map to track what we expect the cache stack to represent.
112  std::map<uint256, CCoins> result;
113 
114  // The cache stack.
115  CCoinsViewTest base; // A CCoinsViewTest at the bottom.
116  std::vector<CCoinsViewCacheTest*> stack; // A stack of CCoinsViewCaches on top.
117  stack.push_back(new CCoinsViewCacheTest(&base)); // Start with one cache.
118 
119  // Use a limited set of random transaction ids, so we do test overwriting entries.
120  std::vector<uint256> txids;
121  txids.resize(NUM_SIMULATION_ITERATIONS / 8);
122  for (unsigned int i = 0; i < txids.size(); i++) {
123  txids[i] = GetRandHash();
124  }
125 
126  for (unsigned int i = 0; i < NUM_SIMULATION_ITERATIONS; i++) {
127  // Do a random modification.
128  {
129  uint256 txid = txids[insecure_rand() % txids.size()]; // txid we're going to modify in this iteration.
130  CCoins& coins = result[txid];
131  CCoinsModifier entry = stack.back()->ModifyCoins(txid);
132  BOOST_CHECK(coins == *entry);
133  if (insecure_rand() % 5 == 0 || coins.IsPruned()) {
134  if (coins.IsPruned()) {
135  added_an_entry = true;
136  } else {
137  updated_an_entry = true;
138  }
139  coins.nVersion = insecure_rand();
140  coins.vout.resize(1);
141  coins.vout[0].nValue = insecure_rand();
142  *entry = coins;
143  } else {
144  coins.Clear();
145  entry->Clear();
146  removed_an_entry = true;
147  }
148  }
149 
150  // Once every 1000 iterations and at the end, verify the full cache.
151  if (insecure_rand() % 1000 == 1 || i == NUM_SIMULATION_ITERATIONS - 1) {
152  for (std::map<uint256, CCoins>::iterator it = result.begin(); it != result.end(); it++) {
153  const CCoins* coins = stack.back()->AccessCoins(it->first);
154  if (coins) {
155  BOOST_CHECK(*coins == it->second);
156  found_an_entry = true;
157  } else {
158  BOOST_CHECK(it->second.IsPruned());
159  missed_an_entry = true;
160  }
161  }
162  BOOST_FOREACH(const CCoinsViewCacheTest *test, stack) {
163  test->SelfTest();
164  }
165  }
166 
167  if (insecure_rand() % 100 == 0) {
168  // Every 100 iterations, flush an intermediate cache
169  if (stack.size() > 1 && insecure_rand() % 2 == 0) {
170  unsigned int flushIndex = insecure_rand() % (stack.size() - 1);
171  stack[flushIndex]->Flush();
172  }
173  }
174  if (insecure_rand() % 100 == 0) {
175  // Every 100 iterations, change the cache stack.
176  if (stack.size() > 0 && insecure_rand() % 2 == 0) {
177  //Remove the top cache
178  stack.back()->Flush();
179  delete stack.back();
180  stack.pop_back();
181  }
182  if (stack.size() == 0 || (stack.size() < 4 && insecure_rand() % 2)) {
183  //Add a new cache
184  CCoinsView* tip = &base;
185  if (stack.size() > 0) {
186  tip = stack.back();
187  } else {
188  removed_all_caches = true;
189  }
190  stack.push_back(new CCoinsViewCacheTest(tip));
191  if (stack.size() == 4) {
192  reached_4_caches = true;
193  }
194  }
195  }
196  }
197 
198  // Clean up the stack.
199  while (stack.size() > 0) {
200  delete stack.back();
201  stack.pop_back();
202  }
203 
204  // Verify coverage.
205  BOOST_CHECK(removed_all_caches);
206  BOOST_CHECK(reached_4_caches);
207  BOOST_CHECK(added_an_entry);
208  BOOST_CHECK(removed_an_entry);
209  BOOST_CHECK(updated_an_entry);
210  BOOST_CHECK(found_an_entry);
211  BOOST_CHECK(missed_an_entry);
212 }

214 // This test is similar to the previous test
215 // except the emphasis is on testing the functionality of UpdateCoins
216 // random txs are created and UpdateCoins is used to update the cache stack
217 // In particular it is tested that spending a duplicate coinbase tx
218 // has the expected effect (the other duplicate is overwitten at all cache levels)
219 BOOST_AUTO_TEST_CASE(updatecoins_simulation_test)
220 {
221  bool spent_a_duplicate_coinbase = false;
222  // A simple map to track what we expect the cache stack to represent.
223  std::map<uint256, CCoins> result;
224 
225  // The cache stack.
226  CCoinsViewTest base; // A CCoinsViewTest at the bottom.
227  std::vector<CCoinsViewCacheTest*> stack; // A stack of CCoinsViewCaches on top.
228  stack.push_back(new CCoinsViewCacheTest(&base)); // Start with one cache.
229 
230  // Track the txids we've used and whether they have been spent or not
231  std::map<uint256, CAmount> coinbaseids;
232  std::set<uint256> alltxids;
233  std::set<uint256> duplicateids;
234 
235  for (unsigned int i = 0; i < NUM_SIMULATION_ITERATIONS; i++) {
236  {
237  CMutableTransaction tx;
238  tx.vin.resize(1);
239  tx.vout.resize(1);
240  tx.vout[0].nValue = i; //Keep txs unique unless intended to duplicate
241  unsigned int height = insecure_rand();
242 
243  // 1/10 times create a coinbase
244  if (insecure_rand() % 10 == 0 || coinbaseids.size() < 10) {
245  // 1/100 times create a duplicate coinbase
246  if (insecure_rand() % 10 == 0 && coinbaseids.size()) {
247  std::map<uint256, CAmount>::iterator coinbaseIt = coinbaseids.lower_bound(GetRandHash());
248  if (coinbaseIt == coinbaseids.end()) {
249  coinbaseIt = coinbaseids.begin();
250  }
251  //Use same random value to have same hash and be a true duplicate
252  tx.vout[0].nValue = coinbaseIt->second;
253  assert(tx.GetHash() == coinbaseIt->first);
254  duplicateids.insert(coinbaseIt->first);
255  }
256  else {
257  coinbaseids[tx.GetHash()] = tx.vout[0].nValue;
258  }
259  assert(CTransaction(tx).IsCoinBase());
260  }
261  // 9/10 times create a regular tx
262  else {
263  uint256 prevouthash;
264  // equally likely to spend coinbase or non coinbase
265  std::set<uint256>::iterator txIt = alltxids.lower_bound(GetRandHash());
266  if (txIt == alltxids.end()) {
267  txIt = alltxids.begin();
268  }
269  prevouthash = *txIt;
270 
271  // Construct the tx to spend the coins of prevouthash
272  tx.vin[0].prevout.hash = prevouthash;
273  tx.vin[0].prevout.n = 0;
274 
275  // Update the expected result of prevouthash to know these coins are spent
276  CCoins& oldcoins = result[prevouthash];
277  oldcoins.Clear();
278 
279  // It is of particular importance here that once we spend a coinbase tx hash
280  // it is no longer available to be duplicated (or spent again)
281  // BIP 34 in conjunction with enforcing BIP 30 (at least until BIP 34 was active)
282  // results in the fact that no coinbases were duplicated after they were already spent
283  alltxids.erase(prevouthash);
284  coinbaseids.erase(prevouthash);
285 
286  // The test is designed to ensure spending a duplicate coinbase will work properly
287  // if that ever happens and not resurrect the previously overwritten coinbase
288  if (duplicateids.count(prevouthash))
289  spent_a_duplicate_coinbase = true;
290 
291  assert(!CTransaction(tx).IsCoinBase());
292  }
293  // Track this tx to possibly spend later
294  alltxids.insert(tx.GetHash());
295 
296  // Update the expected result to know about the new output coins
297  CCoins &coins = result[tx.GetHash()];
298  coins.FromTx(tx, height);
299 
300  UpdateCoins(tx, *(stack.back()), height);
301  }
302 
303  // Once every 1000 iterations and at the end, verify the full cache.
304  if (insecure_rand() % 1000 == 1 || i == NUM_SIMULATION_ITERATIONS - 1) {
305  for (std::map<uint256, CCoins>::iterator it = result.begin(); it != result.end(); it++) {
306  const CCoins* coins = stack.back()->AccessCoins(it->first);
307  if (coins) {
308  BOOST_CHECK(*coins == it->second);
309  } else {
310  BOOST_CHECK(it->second.IsPruned());
311  }
312  }
313  }
314 
315  if (insecure_rand() % 100 == 0) {
316  // Every 100 iterations, flush an intermediate cache
317  if (stack.size() > 1 && insecure_rand() % 2 == 0) {
318  unsigned int flushIndex = insecure_rand() % (stack.size() - 1);
319  stack[flushIndex]->Flush();
320  }
321  }
322  if (insecure_rand() % 100 == 0) {
323  // Every 100 iterations, change the cache stack.
324  if (stack.size() > 0 && insecure_rand() % 2 == 0) {
325  stack.back()->Flush();
326  delete stack.back();
327  stack.pop_back();
328  }
329  if (stack.size() == 0 || (stack.size() < 4 && insecure_rand() % 2)) {
330  CCoinsView* tip = &base;
331  if (stack.size() > 0) {
332  tip = stack.back();
333  }
334  stack.push_back(new CCoinsViewCacheTest(tip));
335  }
336  }
337  }
338 
339  // Clean up the stack.
340  while (stack.size() > 0) {
341  delete stack.back();
342  stack.pop_back();
343  }
344 
345  // Verify coverage.
346  BOOST_CHECK(spent_a_duplicate_coinbase);
347 }
