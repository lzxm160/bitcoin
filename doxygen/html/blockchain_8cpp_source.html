46 double GetDifficulty(const CBlockIndex* blockindex)
47 {
48  // Floating point number that is a multiple of the minimum difficulty,
49  // minimum difficulty = 1.0.
50  if (blockindex == NULL)
51  {
52  if (chainActive.Tip() == NULL)
53  return 1.0;
54  else
55  blockindex = chainActive.Tip();
56  }
57 
58  int nShift = (blockindex->nBits >> 24) & 0xff;
59 
60  double dDiff =
61  (double)0x0000ffff / (double)(blockindex->nBits & 0x00ffffff);
62 
63  while (nShift < 29)
64  {
65  dDiff *= 256.0;
66  nShift++;
67  }
68  while (nShift > 29)
69  {
70  dDiff /= 256.0;
71  nShift--;
72  }
73 
74  return dDiff;
75 }

777 static bool GetUTXOStats(CCoinsView *view, CCoinsStats &stats)
778 {
779  std::unique_ptr<CCoinsViewCursor> pcursor(view->Cursor());
780 
781  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
782  stats.hashBlock = pcursor->GetBestBlock();
783  {
784  LOCK(cs_main);
785  stats.nHeight = mapBlockIndex.find(stats.hashBlock)->second->nHeight;
786  }
787  ss << stats.hashBlock;
788  CAmount nTotalAmount = 0;
789  while (pcursor->Valid()) {
790  boost::this_thread::interruption_point();
791  uint256 key;
792  CCoins coins;
793  if (pcursor->GetKey(key) && pcursor->GetValue(coins)) {
794  stats.nTransactions++;
795  ss << key;
796  for (unsigned int i=0; i<coins.vout.size(); i++) {
797  const CTxOut &out = coins.vout[i];
798  if (!out.IsNull()) {
799  stats.nTransactionOutputs++;
800  ss << VARINT(i+1);
801  ss << out;
802  nTotalAmount += out.nValue;
803  }
804  }
805  stats.nSerializedSize += 32 + pcursor->GetValueSize();
806  ss << VARINT(0);
807  } else {
808  return error("%s: unable to read value", __func__);
809  }
810  pcursor->Next();
811  }
812  stats.hashSerialized = ss.GetHash();
813  stats.nTotalAmount = nTotalAmount;
814  return true;
