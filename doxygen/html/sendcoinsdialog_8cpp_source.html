191 void SendCoinsDialog::on_sendButton_clicked()
192 {
193  if(!model || !model->getOptionsModel())
194  return;
195 
196  QList<SendCoinsRecipient> recipients;
197  bool valid = true;
198 
199  for(int i = 0; i < ui->entries->count(); ++i)
200  {
201  SendCoinsEntry *entry = qobject_cast<SendCoinsEntry*>(ui->entries->itemAt(i)->widget());
202  if(entry)
203  {
204  if(entry->validate())
205  {
206  recipients.append(entry->getValue());
207  }
208  else
209  {
210  valid = false;
211  }
212  }
213  }
214 
215  if(!valid || recipients.isEmpty())
216  {
217  return;
218  }
219 
220  fNewRecipientAllowed = false;
221  WalletModel::UnlockContext ctx(model->requestUnlock());
222  if(!ctx.isValid())
223  {
224  // Unlock wallet was cancelled
225  fNewRecipientAllowed = true;
226  return;
227  }
228 
229  // prepare transaction for getting txFee earlier
230  WalletModelTransaction currentTransaction(recipients);
231  WalletModel::SendCoinsReturn prepareStatus;
232  if (model->getOptionsModel()->getCoinControlFeatures()) // coin control enabled
233  prepareStatus = model->prepareTransaction(currentTransaction, CoinControlDialog::coinControl);
234  else
235  prepareStatus = model->prepareTransaction(currentTransaction);
236 
237  // process prepareStatus and on error generate message shown to user
238  processSendCoinsReturn(prepareStatus,
239  BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), currentTransaction.getTransactionFee()));
240 
241  if(prepareStatus.status != WalletModel::OK) {
242  fNewRecipientAllowed = true;
243  return;
244  }
245 
246  CAmount txFee = currentTransaction.getTransactionFee();
247 
248  // Format confirmation message
249  QStringList formatted;
250  Q_FOREACH(const SendCoinsRecipient &rcp, currentTransaction.getRecipients())
251  {
252  // generate bold amount string
253  QString amount = "<b>" + BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), rcp.amount);
254  amount.append("</b>");
255  // generate monospace address string
256  QString address = "<span style='font-family: monospace;'>" + rcp.address;
257  address.append("</span>");
258 
259  QString recipientElement;
260 
261  if (!rcp.paymentRequest.IsInitialized()) // normal payment
262  {
263  if(rcp.label.length() > 0) // label with address
264  {
265  recipientElement = tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.label));
266  recipientElement.append(QString(" (%1)").arg(address));
267  }
268  else // just address
269  {
270  recipientElement = tr("%1 to %2").arg(amount, address);
271  }
272  }
273  else if(!rcp.authenticatedMerchant.isEmpty()) // authenticated payment request
274  {
275  recipientElement = tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.authenticatedMerchant));
276  }
277  else // unauthenticated payment request
278  {
279  recipientElement = tr("%1 to %2").arg(amount, address);
280  }
281 
282  formatted.append(recipientElement);
283  }
284 
285  QString questionString = tr("Are you sure you want to send?");
286  questionString.append("<br /><br />%1");
287 
288  if(txFee > 0)
289  {
290  // append fee string if a fee is required
291  questionString.append("<hr /><span style='color:#aa0000;'>");
292  questionString.append(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), txFee));
293  questionString.append("</span> ");
294  questionString.append(tr("added as transaction fee"));
295 
296  // append transaction size
297  questionString.append(" (" + QString::number((double)currentTransaction.getTransactionSize() / 1000) + " kB)");
298  }
299 
300  // add total amount in all subdivision units
301  questionString.append("<hr />");
302  CAmount totalAmount = currentTransaction.getTotalTransactionAmount() + txFee;
303  QStringList alternativeUnits;
304  Q_FOREACH(BitcoinUnits::Unit u, BitcoinUnits::availableUnits())
305  {
306  if(u != model->getOptionsModel()->getDisplayUnit())
307  alternativeUnits.append(BitcoinUnits::formatHtmlWithUnit(u, totalAmount));
308  }
309  questionString.append(tr("Total Amount %1")
310  .arg(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), totalAmount)));
311  questionString.append(QString("<span style='font-size:10pt;font-weight:normal;'><br />(=%2)</span>")
312  .arg(alternativeUnits.join(" " + tr("or") + "<br />")));
313 
314  SendConfirmationDialog confirmationDialog(tr("Confirm send coins"),
315  questionString.arg(formatted.join("<br />")), SEND_CONFIRM_DELAY, this);
316  confirmationDialog.exec();
317  QMessageBox::StandardButton retval = (QMessageBox::StandardButton)confirmationDialog.result();
318 
319  if(retval != QMessageBox::Yes)
320  {
321  fNewRecipientAllowed = true;
322  return;
323  }
324 
325  // now send the prepared transaction
326  WalletModel::SendCoinsReturn sendStatus = model->sendCoins(currentTransaction);
327  // process sendStatus and on error generate message shown to user
328  processSendCoinsReturn(sendStatus);
329 
330  if (sendStatus.status == WalletModel::OK)
331  {
332  accept();
333  CoinControlDialog::coinControl->UnSelectAll();
334  coinControlUpdateLabels();
335  }
336  fNewRecipientAllowed = true;
337 }

493 void SendCoinsDialog::processSendCoinsReturn(const WalletModel::SendCoinsReturn &sendCoinsReturn, const QString &msgArg)
494 {
495  QPair<QString, CClientUIInterface::MessageBoxFlags> msgParams;
496  // Default to a warning message, override if error message is needed
497  msgParams.second = CClientUIInterface::MSG_WARNING;
498 
499  // This comment is specific to SendCoinsDialog usage of WalletModel::SendCoinsReturn.
500  // WalletModel::TransactionCommitFailed is used only in WalletModel::sendCoins()
501  // all others are used only in WalletModel::prepareTransaction()
502  switch(sendCoinsReturn.status)
503  {
504  case WalletModel::InvalidAddress:
505  msgParams.first = tr("The recipient address is not valid. Please recheck.");
506  break;
507  case WalletModel::InvalidAmount:
508  msgParams.first = tr("The amount to pay must be larger than 0.");
509  break;
510  case WalletModel::AmountExceedsBalance:
511  msgParams.first = tr("The amount exceeds your balance.");
512  break;
513  case WalletModel::AmountWithFeeExceedsBalance:
514  msgParams.first = tr("The total exceeds your balance when the %1 transaction fee is included.").arg(msgArg);
515  break;
516  case WalletModel::DuplicateAddress:
517  msgParams.first = tr("Duplicate address found: addresses should only be used once each.");
518  break;
519  case WalletModel::TransactionCreationFailed:
520  msgParams.first = tr("Transaction creation failed!");
521  msgParams.second = CClientUIInterface::MSG_ERROR;
522  break;
523  case WalletModel::TransactionCommitFailed:
524  msgParams.first = tr("The transaction was rejected! This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
525  msgParams.second = CClientUIInterface::MSG_ERROR;
526  break;
527  case WalletModel::AbsurdFee:
528  msgParams.first = tr("A fee higher than %1 is considered an absurdly high fee.").arg(BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), maxTxFee));
529  break;
530  case WalletModel::PaymentRequestExpired:
531  msgParams.first = tr("Payment request expired.");
532  msgParams.second = CClientUIInterface::MSG_ERROR;
533  break;
534  // included to prevent a compiler warning.
535  case WalletModel::OK:
536  default:
537  return;
538  }
539 
540  Q_EMIT message(tr("Send Coins"), msgParams.first, msgParams.second);
541 }
