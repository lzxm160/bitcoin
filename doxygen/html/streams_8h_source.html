340  void Xor(const std::vector<unsigned char>& key)
341  {
342  if (key.size() == 0) {
343  return;
344  }
345 
346  for (size_type i = 0, j = 0; i != size(); i++) {
347  vch[i] ^= key[j++];
348 
349  // This potentially acts on very many bytes of data, so it's
350  // important that we calculate `j`, i.e. the `key` index in this
351  // way instead of doing a %, which would effectively be a division
352  // for each byte Xor'd -- much slower than need be.
353  if (j == key.size())
354  j = 0;
355  }
356  }

518  bool Fill() {
519  unsigned int pos = nSrcPos % vchBuf.size();
520  unsigned int readNow = vchBuf.size() - pos;
521  unsigned int nAvail = vchBuf.size() - (nSrcPos - nReadPos) - nRewind;

529  } else {
530  nSrcPos += read;
531  return true;
532  }
