116 unsigned int CTransaction::CalculateModifiedSize(unsigned int nTxSize) const
117 {
118  // In order to avoid disincentivizing cleaning up the UTXO set we don't count
119  // the constant overhead for each txin and up to 110 bytes of scriptSig (which
120  // is enough to cover a compressed pubkey p2sh redemption) for priority.
121  // Providing any more cleanup incentive than making additional inputs free would
122  // risk encouraging people to create junk outputs to redeem later.
123  if (nTxSize == 0)
124  nTxSize = (GetTransactionWeight(*this) + WITNESS_SCALE_FACTOR - 1) / WITNESS_SCALE_FACTOR;
125  for (std::vector<CTxIn>::const_iterator it(vin.begin()); it != vin.end(); ++it)
126  {
127  unsigned int offset = 41U + std::min(110U, (unsigned int)it->scriptSig.size());
128  if (nTxSize > offset)
129  nTxSize -= offset;
130  }
131  return nTxSize;
132 }
