49 bool CCoinsViewDB::BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlock) {
50  CDBBatch batch(db);
51  size_t count = 0;
52  size_t changed = 0;
53  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
54  if (it->second.flags & CCoinsCacheEntry::DIRTY) {
55  if (it->second.coins.IsPruned())
56  batch.Erase(make_pair(DB_COINS, it->first));
57  else
58  batch.Write(make_pair(DB_COINS, it->first), it->second.coins);
59  changed++;
60  }
61  count++;
62  CCoinsMap::iterator itOld = it++;
63  mapCoins.erase(itOld);
64  }
65  if (!hashBlock.IsNull())
66  batch.Write(DB_BEST_BLOCK, hashBlock);
67 
68  LogPrint("coindb", "Committing %u changed transactions (out of %u) to coin database...\n", (unsigned int)changed, (unsigned int)count);
69  return db.WriteBatch(batch);
70 }

175 {
176  std::unique_ptr<CDBIterator> pcursor(NewIterator());

178  pcursor->Seek(make_pair(DB_BLOCK_INDEX, uint256()));
179 
180  // Load mapBlockIndex

182  boost::this_thread::interruption_point();
