225 class CompareTxMemPoolEntryByDescendantScore
226 {
227 public:
228  bool operator()(const CTxMemPoolEntry& a, const CTxMemPoolEntry& b)
229  {
230  bool fUseADescendants = UseDescendantScore(a);
231  bool fUseBDescendants = UseDescendantScore(b);
232 
233  double aModFee = fUseADescendants ? a.GetModFeesWithDescendants() : a.GetModifiedFee();
234  double aSize = fUseADescendants ? a.GetSizeWithDescendants() : a.GetTxSize();
235 
236  double bModFee = fUseBDescendants ? b.GetModFeesWithDescendants() : b.GetModifiedFee();
237  double bSize = fUseBDescendants ? b.GetSizeWithDescendants() : b.GetTxSize();
238 
239  // Avoid division by rewriting (a/b > c/d) as (a*d > c*b).
240  double f1 = aModFee * bSize;
241  double f2 = aSize * bModFee;
242 
243  if (f1 == f2) {
244  return a.GetTime() >= b.GetTime();
245  }
246  return f1 < f2;
247  }
248 
249  // Calculate which score to use for an entry (avoiding division).
250  bool UseDescendantScore(const CTxMemPoolEntry &a)
251  {
252  double f1 = (double)a.GetModifiedFee() * a.GetSizeWithDescendants();
253  double f2 = (double)a.GetModFeesWithDescendants() * a.GetTxSize();
254  return f2 > f1;
255  }
256 };
