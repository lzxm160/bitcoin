247 static void BlockTipChanged(ClientModel *clientmodel, bool initialSync, const CBlockIndex *pIndex, bool fHeader)
248 {
249  // lock free async UI updates in case we have a new block tip
250  // during initial sync, only update the UI if the last update
251  // was > 250ms (MODEL_UPDATE_DELAY) ago
252  int64_t now = 0;
253  if (initialSync)
254  now = GetTimeMillis();
255 
256  int64_t& nLastUpdateNotification = fHeader ? nLastHeaderTipUpdateNotification : nLastBlockTipUpdateNotification;
257 
258  // if we are in-sync, update the UI regardless of last update time
259  if (!initialSync || now - nLastUpdateNotification > MODEL_UPDATE_DELAY) {
260  //pass a async signal to the UI thread
261  QMetaObject::invokeMethod(clientmodel, "numBlocksChanged", Qt::QueuedConnection,
262  Q_ARG(int, pIndex->nHeight),
263  Q_ARG(QDateTime, QDateTime::fromTime_t(pIndex->GetBlockTime())),
264  Q_ARG(double, clientmodel->getVerificationProgress(pIndex)),
265  Q_ARG(bool, fHeader));
266  nLastUpdateNotification = now;
267  }
268 }
