240 inline unsigned int GetSizeOfCompactSize(uint64_t nSize)
241 {
242  if (nSize < 253) return sizeof(unsigned char);
243  else if (nSize <= std::numeric_limits<unsigned short>::max()) return sizeof(unsigned char) + sizeof(unsigned short);
244  else if (nSize <= std::numeric_limits<unsigned int>::max()) return sizeof(unsigned char) + sizeof(unsigned int);
245  else return sizeof(unsigned char) + sizeof(uint64_t);
246 }

248 template<typename Stream>
249 void WriteCompactSize(Stream& os, uint64_t nSize)
250 {
251  if (nSize < 253)
252  {
253  ser_writedata8(os, nSize);
254  }
255  else if (nSize <= std::numeric_limits<unsigned short>::max())
256  {
257  ser_writedata8(os, 253);
258  ser_writedata16(os, nSize);
259  }
260  else if (nSize <= std::numeric_limits<unsigned int>::max())
261  {
262  ser_writedata8(os, 254);
263  ser_writedata32(os, nSize);
264  }
265  else
266  {
267  ser_writedata8(os, 255);
268  ser_writedata64(os, nSize);
269  }
270  return;
271 }

273 template<typename Stream>
274 uint64_t ReadCompactSize(Stream& is)
275 {
276  uint8_t chSize = ser_readdata8(is);
277  uint64_t nSizeRet = 0;
278  if (chSize < 253)
279  {
280  nSizeRet = chSize;
281  }
282  else if (chSize == 253)
283  {
284  nSizeRet = ser_readdata16(is);
285  if (nSizeRet < 253)
286  throw std::ios_base::failure("non-canonical ReadCompactSize()");
287  }
288  else if (chSize == 254)
289  {
290  nSizeRet = ser_readdata32(is);
291  if (nSizeRet < 0x10000u)
292  throw std::ios_base::failure("non-canonical ReadCompactSize()");
293  }
294  else
295  {
296  nSizeRet = ser_readdata64(is);
297  if (nSizeRet < 0x100000000ULL)
298  throw std::ios_base::failure("non-canonical ReadCompactSize()");
299  }
300  if (nSizeRet > (uint64_t)MAX_SIZE)
301  throw std::ios_base::failure("ReadCompactSize(): size too large");
302  return nSizeRet;
303 }

342 template<typename Stream, typename I>
343 void WriteVarInt(Stream& os, I n)
344 {
345  unsigned char tmp[(sizeof(n)*8+6)/7];
346  int len=0;
347  while(true) {
348  tmp[len] = (n & 0x7F) | (len ? 0x80 : 0x00);
349  if (n <= 0x7F)
350  break;
351  n = (n >> 7) - 1;
352  len++;
353  }
354  do {
355  ser_writedata8(os, tmp[len]);
356  } while(len--);
357 }

359 template<typename Stream, typename I>
360 I ReadVarInt(Stream& is)
361 {
362  I n = 0;
363  while(true) {
364  unsigned char chData = ser_readdata8(is);

367  n++;
368  else
369  return n;
