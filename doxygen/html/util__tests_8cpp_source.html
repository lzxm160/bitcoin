244 BOOST_AUTO_TEST_CASE(util_seed_insecure_rand)
245 {
246  seed_insecure_rand(true);
247  for (int mod=2;mod<11;mod++)
248  {
249  int mask = 1;
250  // Really rough binomal confidence approximation.
251  int err = 30*10000./mod*sqrt((1./mod*(1-1./mod))/10000.);
252  //mask is 2^ceil(log2(mod))-1
253  while(mask<mod-1)mask=(mask<<1)+1;
254 
255  int count = 0;
256  //How often does it get a zero from the uniform range [0,mod)?
257  for (int i = 0; i < 10000; i++) {
258  uint32_t rval;
259  do{
260  rval=insecure_rand()&mask;
261  }while(rval>=(uint32_t)mod);
262  count += rval==0;
263  }
264  BOOST_CHECK(count<=10000/mod+err);
265  BOOST_CHECK(count>=10000/mod-err);
266  }
267 }
