165  CAmount GetDustThreshold(const CFeeRate &minRelayTxFee) const
166  {
167  // "Dust" is defined in terms of CTransaction::minRelayTxFee,
168  // which has units satoshis-per-kilobyte.
169  // If you'd pay more than 1/3 in fees
170  // to spend something, then we consider it dust.
171  // A typical spendable non-segwit txout is 34 bytes big, and will
172  // need a CTxIn of at least 148 bytes to spend:
173  // so dust is a spendable txout less than
174  // 546*minRelayTxFee/1000 (in satoshis).
175  // A typical spendable segwit txout is 31 bytes big, and will
176  // need a CTxIn of at least 67 bytes to spend:
177  // so dust is a spendable txout less than
178  // 294*minRelayTxFee/1000 (in satoshis).
179  if (scriptPubKey.IsUnspendable())
180  return 0;
181 
182  size_t nSize = GetSerializeSize(SER_DISK, 0);
183  int witnessversion = 0;
184  std::vector<unsigned char> witnessprogram;
185 
186  if (scriptPubKey.IsWitnessProgram(witnessversion, witnessprogram)) {
187  // sum the sizes of the parts of a transaction input
188  // with 75% segwit discount applied to the script size.
189  nSize += (32 + 4 + 1 + (107 / WITNESS_SCALE_FACTOR) + 4);
190  } else {
191  nSize += (32 + 4 + 1 + 107 + 4); // the 148 mentioned above
192  }
193 
194  return 3 * minRelayTxFee.GetFee(nSize);
195  }
196 
197  bool IsDust(const CFeeRate &minRelayTxFee) const
198  {
199  return (nValue < GetDustThreshold(minRelayTxFee));
200  }
201

291 template<typename Stream, typename Operation, typename TxType>
292 inline void SerializeTransaction(TxType& tx, Stream& s, Operation ser_action, int nType, int nVersion) {
293  const bool fAllowWitness = !(nVersion & SERIALIZE_TRANSACTION_NO_WITNESS);
294 
295  READWRITE(*const_cast<int32_t*>(&tx.nVersion));
296  unsigned char flags = 0;
297  if (ser_action.ForRead()) {
298  const_cast<std::vector<CTxIn>*>(&tx.vin)->clear();
299  const_cast<std::vector<CTxOut>*>(&tx.vout)->clear();
300  const_cast<CTxWitness*>(&tx.wit)->SetNull();
301  /* Try to read the vin. In case the dummy is there, this will be read as an empty vector. */
302  READWRITE(*const_cast<std::vector<CTxIn>*>(&tx.vin));
303  if (tx.vin.size() == 0 && fAllowWitness) {
304  /* We read a dummy or an empty vin. */
305  READWRITE(flags);
306  if (flags != 0) {
307  READWRITE(*const_cast<std::vector<CTxIn>*>(&tx.vin));
308  READWRITE(*const_cast<std::vector<CTxOut>*>(&tx.vout));
309  }
310  } else {
311  /* We read a non-empty vin. Assume a normal vout follows. */
312  READWRITE(*const_cast<std::vector<CTxOut>*>(&tx.vout));
313  }
314  if ((flags & 1) && fAllowWitness) {
315  /* The witness flag is present, and we support witnesses. */
316  flags ^= 1;
317  const_cast<CTxWitness*>(&tx.wit)->vtxinwit.resize(tx.vin.size());
318  READWRITE(tx.wit);
319  }
320  if (flags) {
321  /* Unknown flag in the serialization */
322  throw std::ios_base::failure("Unknown transaction optional data");
323  }
324  } else {
325  // Consistency check
326  assert(tx.wit.vtxinwit.size() <= tx.vin.size());
327  if (fAllowWitness) {
328  /* Check whether witnesses need to be serialized. */
329  if (!tx.wit.IsNull()) {
330  flags |= 1;
331  }
332  }
333  if (flags) {
334  /* Use extended format in case witnesses are to be serialized. */
335  std::vector<CTxIn> vinDummy;
336  READWRITE(vinDummy);
337  READWRITE(flags);
338  }
339  READWRITE(*const_cast<std::vector<CTxIn>*>(&tx.vin));
340  READWRITE(*const_cast<std::vector<CTxOut>*>(&tx.vout));
341  if (flags & 1) {
342  const_cast<CTxWitness*>(&tx.wit)->vtxinwit.resize(tx.vin.size());
343  READWRITE(tx.wit);
344  }
345  }
346  READWRITE(*const_cast<uint32_t*>(&tx.nLockTime));
347 }
